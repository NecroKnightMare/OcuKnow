cmake_minimum_required(VERSION 3.16)
project(OcuKnow)

set(CMAKE_CXX_STANDARD 17)

find_package(Qt6 REQUIRED COMPONENTS Core Gui Widgets Multimedia)

qt_add_resources(OcuKnowResources PREFIX "/" FILES
	src/graphics/ocuknow_intro.gif
)

add_executable(OcuKnow
	main.cpp
	UI/splashscreen.cpp
	UI/intro.h
	${OcuKnowResources}
)

target_link_libraries(OcuKnow Qt6::Core Qt6::Gui Qt6::Widgets Qt6::Multimedia)
